204#define C4LogToAt(DOMAIN, LEVEL, FMT, ...) \
205 do { \
206 if ( c4log_willLog(DOMAIN, LEVEL) ) c4log(DOMAIN, LEVEL, FMT, ##__VA_ARGS__); \
207 } while ( false )
