16 // Non-null annotations, for function parameters and struct fields.
17 // In between C4_ASSUME_NONNULL_BEGIN and C4_ASSUME_NONNULL_END, all pointer declarations implicitly
18 // disallow NULL values, unless annotated with C4NULLABLE (which must come after the `*`.)
19 // (C4NONNULL is occasionally necessary when there are multiple levels of pointers.)
20 // NOTE: Does not apply to function return values, for some reason. Those may still be null,
21 // unless annotated with C4_RETURNS_NONNULL.
22 // NOTE: Only supported in Clang, so far.
23 #if __has_feature(nullability)
24 # define C4_ASSUME_NONNULL_BEGIN _Pragma("clang assume_nonnull begin")
25 # define C4_ASSUME_NONNULL_END _Pragma("clang assume_nonnull end")
26 # define C4NULLABLE _Nullable
27 # define C4NONNULL _Nonnull
28 # define C4_RETURNS_NONNULL __attribute__((returns_nonnull))

30 # define C4_ASSUME_NONNULL_BEGIN

33 # define C4NONNULL
